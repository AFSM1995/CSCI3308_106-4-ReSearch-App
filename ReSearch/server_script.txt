Server Scripts for creating ReSearch database and profile table
PostgreSQL server stuff will be stored in documents like this until we learn how to store a database server


Postgres Log-in
    sudo -u postgres psql


Creating database

 create database research_db;

Enter database

 \c research_db;

Creating user_profiles table

CREATE TABLE IF NOT EXISTS user_profiles (
    id SERIAL PRIMARY KEY, /* Unique identifier for each user */
    name VARCHAR(50) NOT NULL, /* user's first and last name */
    email VARCHAR(100) NOT NULL,
    username VARCHAR(30) NOT NULL,
    type VARCHAR(1) NOT NULL, /* Profile type: U for user, R for researcher, D for dual */
    password VARCHAR(30) NOT NULL,
    birthday DATE,
    year VARCHAR(3)
);

Creating user_data table

CREATE TABLE IF NOT EXISTS user_data (
    encryptId TEXT NOT NULL PRIMARY KEY,
    major VARCHAR(4),
    psychReport TEXT,
    demographics TEXT,
    medHistory TEXT
);

Creating researcher_data table

CREATE TABLE IF NOT EXISTS researcher_data (
    encryptId TEXT NOT NULL PRIMARY KEY,
    projectName VARCHAR(50),
    credentials TEXT,
    field VARCHAR(4),
    institution VARCHAR(50)
);

Creating majors table

CREATE TABLE IF NOT EXISTS majors (
    majorCode VARCHAR(4) NOT NULL PRIMARY KEY,
    majorDescription TEXT NOT NULL,
    numEnrolled INT
);

Creating postings table

CREATE TABLE IF NOT EXISTS postings (
    title VARCHAR(50) NOT NULL,
    school VARCHAR(50) NOT NULL,
    city VARCHAR(40) NOT NULL,
    state VARCHAR(2) NOT NULL,
    zip NUMERIC(5) NOT NULL,
    body TEXT NOT NULL,
    major VARCHAR(40),
    app_open DATE,
    app_close DATE,
    start_date DATE,
    end_date DATE,
    contact_name VARCHAR(50),
    contact_email VARCHAR(50),
    contact_phone VARCHAR(15),
    contact_fax VARCHAR(20)
);
